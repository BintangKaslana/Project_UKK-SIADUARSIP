require_once __DIR__ . '/public/connection.php';

$steps = [];

try {
    // 1. Tambah kolom role ke tabel admin (head_admin / admin)
    $conn->exec("ALTER TABLE admin ADD COLUMN IF NOT EXISTS role VARCHAR(20) NOT NULL DEFAULT 'admin' CHECK (role IN ('head_admin','admin'))");
    $steps[] = "✅ Kolom 'role' berhasil ditambahkan ke tabel admin.";
} catch (Exception $e) {
    $steps[] = "⚠️ Kolom 'role': " . $e->getMessage();
}

try {
    // 2. Tambah kolom full_name ke tabel admin
    $conn->exec("ALTER TABLE admin ADD COLUMN IF NOT EXISTS full_name VARCHAR(100) DEFAULT NULL");
    $steps[] = "✅ Kolom 'full_name' berhasil ditambahkan ke tabel admin.";
} catch (Exception $e) {
    $steps[] = "⚠️ Kolom 'full_name': " . $e->getMessage();
}

try {
    // 3. Set admin pertama (id=1) sebagai head_admin
    $conn->exec("UPDATE admin SET role = 'head_admin' WHERE id = (SELECT MIN(id) FROM admin)");
    $steps[] = "✅ Admin pertama di-set sebagai head_admin.";
} catch (Exception $e) {
    $steps[] = "⚠️ Update role head_admin: " . $e->getMessage();
}

try {
    // 4. Tambah kolom feedback_by ke tabel aspirasi (menyimpan admin_id yang memberi feedback)
    $conn->exec("ALTER TABLE aspirasi ADD COLUMN IF NOT EXISTS feedback_by INTEGER DEFAULT NULL REFERENCES admin(id) ON DELETE SET NULL");
    $steps[] = "✅ Kolom 'feedback_by' berhasil ditambahkan ke tabel aspirasi.";
} catch (Exception $e) {
    $steps[] = "⚠️ Kolom 'feedback_by': " . $e->getMessage();
}

try {
    // 5. Tambah kolom reviewed_by ke tabel aspirasi (admin yang mereview/approve/reject)
    $conn->exec("ALTER TABLE aspirasi ADD COLUMN IF NOT EXISTS reviewed_by INTEGER DEFAULT NULL REFERENCES admin(id) ON DELETE SET NULL");
    $steps[] = "✅ Kolom 'reviewed_by' berhasil ditambahkan ke tabel aspirasi.";
} catch (Exception $e) {
    $steps[] = "⚠️ Kolom 'reviewed_by': " . $e->getMessage();
}
